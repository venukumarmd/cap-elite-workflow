 /* Node is defined as :
 class Node 
    int data;
    Node left;
    Node right;
    
    */

	public static Node insert(Node root,int data) {
        if(root == null){
            Node temp = new Node(data);
            return temp;
        }
        Node parent = root;
        Node curr = root;
        while(curr != null){
            parent = curr;
            if(curr.data > data){
                curr = curr.left;
            }
            else{
                curr = curr.right;
            }
        }
        Node temp = new Node(data);
        if(parent.data > data){
            parent.left = temp;
        }
        else{
            parent.right = temp;
        }
        return root;
    }
